from IPython.core.display import display, HTML

def display_results(results, highlight_name='none'):
    htmlstr = ''
    for output in results['outputs']:
        url = output['input']['data']['image']['url']
        htmlstr += f'<figure style="display:inline-block"><a target="_blank" href="{url}">'\
            f'<img src="{url}" width=100 height=100 /></a>\n'
        count = 0
        for _concept in output['data']['concepts']:
            name = _concept['name']
            value = _concept['value']
            highlight=''
            if name == highlight_name or (highlight_name == 'none' and count==0):
                if float(value) > 0.7:
                    highlight='style="background-color:green"'
                elif 0.5 < float(value) <= 0.7:
                    highlight='style="background-color:yellow"'
                elif float(value) <= 0.5:
                    highlight='style="background-color:red"'
                count += 1
            htmlstr += f'<figcaption {highlight}>{name}:{value:.4f}</figcaption>\n'
        htmlstr += '</figure>'
    display(HTML(htmlstr))
